<h5>If/Else control flow</h5>

```swift
if trafficLight == "green" {
    go ()
} else if trafficLight == "amber" {
    slowDown()
} else {
    stop()
}
```

<h5>Swith Statement</h5>

```swift
switch hardness {
    case "Soft":
        print(5)
    case "Medium":
        print(7)
    case "Hard":
        print(12)
    default:
    print("Error")
}
```
